	source_id int identity (1,1) primary key,
	source_name varchar(1000) not null,
	connection_string varchar(1000) not null

CREATE PROCEDURE USP_INS_SOURCE
(
 @SOURCE_NAME VARCHAR(1000),
 @SOURCE_CONN VARCHAR(1000)
)
AS
 BEGIN
  SET NOCOUNT ON;
  SET XACT_ABORT ON;
  INSERT INTO SOURCES (
   source_name,
   connection_string
   )
  VALUES (
   @SOURCE_NAME,
   @SOURCE_CONN
   );
 END
